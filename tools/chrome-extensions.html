        # Chrome-Extensions谷歌浏览器插件

        # mainfest.json

        最基础需要包含以下内容

        {
  // 必须有的配置
  // 只能是2
  "manifest_version": 2,
  // extension名称
  "name": "My Extension",
  // 版本
  "version": "versionString",

  // 建议添加的配置
  // 默认的语言
  "default_locale": "zh",
  // 插件介绍
  "description": "插件介绍",
  // icon
  "icons": {
    "16": "images/get_started16.png",
    "32": "images/get_started32.png",
    "48": "images/get_started48.png",
    "128": "images/get_started128.png"
  },

  // 选择一个或者不选
  // 浏览器右上角图标设置，browser_action、page_action、app必须三选一
  // 显示在所有页面中
	"browser_action": {
        // 默认的图标
		"default_icon": "img/icon.png",
		// 图标悬停时的标题，可选
		"default_title": "这是一个示例Chrome插件",
		"default_popup": "popup.html"
	},
	// 当某些特定页面打开才显示的图标
	/*"page_action":
	{
		"default_icon": "img/icon.png",
		"default_title": "我是pageAction",
		"default_popup": "popup.html"
	},*/
  // Optional
  "action": ...,
  "author": ‘qhj‘,
  "automation": ...,
  "background": {
    // 建议
    "persistent": false,
    // 可选
    "service_worker":
  },
  // 覆盖当前的chrome配置
  "chrome_settings_overrides": {...},
  // 覆盖当前的chrome洁面配置
  "chrome_ui_overrides": {
    "bookmarks_ui": {
      "remove_bookmark_shortcut": true,
      "remove_button": true
    }
  },
  // 修改点击相应动作时返回的页面链接，只支持bookmarks、history、newtab三个页面
  "chrome_url_overrides": {
    "bookmarks": "myPage.html",
    "history": "myPage.html",
    "newtab": "myPage.html"
  },
  // 触发的快捷键
  "commands": {
    "_execute_browser_action": {
      "suggested_key": {
        "windows": "Ctrl+Shift+Y",
        "mac": "Command+Shift+Y",
        "chromeos": "Ctrl+Shift+U",
        "linux": "Ctrl+Shift+J"
      }
    },

  },
  // 页面权限
  "content_capabilities": ...,
   // 直接注入页面的js和css脚本
  "content_scripts": [{
      "matches": ["http://www.google.com/*"],
      "css": ["mystyles.css"],
      "js": ["jquery.js", "myscript.js"],
      // 代码注入的时间，可选值： "document_start", "document_end", or "document_idle"，最后一个表示页面空闲时，默认document_idle
      "run_at": "document_start"
  }],
  // 安全策略，默认情况下禁止使用eval或者Function构造函数，以及内联js，禁止载入外部脚本
  "content_security_policy": "policyString",
  "converted_from_user_script": ...,
  // 当前语言
  "current_locale": ...,
  "declarative_net_request": ...,
  // 在开发中工具中的页面，指向html
  "devtools_page": "devtools.html",
   // 事件监听规则及条件
  "event_rules": [{...}],
  //  哪些外部扩展、应用或网页能连接此插件
  "externally_connectable": {
    "matches": ["*://*.example.com/*"]
  },
  // 允许用户上传文件，只支持Chrome OS
  "file_browser_handlers": [{
    "id": "upload",
    "default_title": "Save to Gallery", // 按钮文字
    "file_filters": [
        "filesystem:*.jpg", // 匹配所有文件可用 "filesystem:*.*"
        "filesystem:*.jpeg",
        "filesystem:*.png"
    ]
  }],
  //  允许访问文件系统，只支持Chrome OS
  "file_system_provider_capabilities": {
    "configurable": true,
    "multiple_mounts": true,
    "source": "network"
  },
  // 主页地址
  "homepage_url": "http://path/to/homepage",
  // 调用其他组件的模块，与其他组件的export属性共用
  "import": [{"id": "aaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaa"}],
  // 隐身模式
  "incognito": "spanning, split, or not_allowed",
  "input_components": ...,
  // 自动生成，可以不需要
  "key": "publicKey",
  "minimum_chrome_version": "versionString",
  "nacl_modules": [...],
  // 谷歌账户信息
  "oauth2": ...,
  // 离线使用
  "offline_enabled": true,
 // 地址栏提供关键字进行搜索
  "omnibox": {
    "keyword": "aString"
  },
  // 运行时权限，安装时无需
  "optional_permissions": ["tabs"],
  // 选项。设置页面，从扩展工具列表进入
  "options_page": "options.html",
  "options_ui": {
    "chrome_style": true,
    "page": "options.html"
  },
  // 安装时提示的权限申请
  "permissions": [
      "contextMenus", // 右键菜单
      "tabs", // 标签
      "notifications", // 通知
      "webRequest", // web请求
      "webRequestBlocking",
      "storage", // 插件本地存储
      "http://*/*", // 可以通过executeScript或者insertCSS访问的网站
      "https://*/*" // 可以通过executeScript或者insertCSS访问的网站
    ],
  "platforms": ...,
  "replacement_web_app": ...,
  // 安装前置需求
  "requirements": {...},
  // 沙盒
  "sandbox": [...],
  // 短名称，最长12个字母
  "short_name": "Short Name",
  "signature": ...,
  "spellcheck": ...,
  "storage": {
    "managed_schema": "schema.json"
  },
  "system_indicator": ...,
  // 文字朗读
  "tts_engine": {
    "voices": [
      {
        "voice_name": "Alice",
        "lang": "en-US",
        "gender": "female",
        "event_types": ["start", "marker", "end"]
      },
      {
        "voice_name": "Pat",
        "lang": "en-US",
        "event_types": ["end"]
      }
    ]
  },
  // 插件更新地址
  "update_url": "http://path/to/updateInfo.xml",
  // 版本名称，和version不同的是，version_name是任意字符串，没有格式要求
  "version_name": "aString",
  // 普通页面能够直接访问的插件资源列表，如果不设置是无法直接访问的
  "web_accessible_resources": [...]
}
}

        # content-scripts

        向页面注入脚本，css和js文件。可在chrome-devtools source面板的conten-scripts查看所有的content-scripts

        脚本加载时间run_at

        • document_start
        • document_end
        • document_idle

        css会影响全局页面

        js:无法访问页面中的js

        # background

        后台常驻页面，浏览器打开后就执行，浏览器关闭后结束。可以指定一个page或者指定script. 能调用的权限非常高

        # popup.html

        弹出页面

        # 插件的形式

        • browserAction
        • pageAction
        • 右键菜单
        • 覆盖特定页面(新标签页newtab、history、bookmarks)
        • option选型页
        • omnibox 提供搜索建议
        • devtools
        • 桌面通知

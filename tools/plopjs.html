    # Plop

    Plop github用于创建模版文件的工具。不仅可以生成静态的文件，

    # 使用

    # 1. 安装plop

    # 安装在项目中
npm install plop --save-dev

# 全局安装
npm install -g plop

    # 2.编写模板文件

    在plop-template文件夹中放置模板文件

    # 3.在项目根目录创建plopfile.js文件

    module.exports = function (plop) {
    // 模板名称
    plop.setGenerator('controller', {
        // 操作的描述
        description: 'application controller',

        // 交互提示
        prompts: [{
            type: 'input',
            name: 'name',
            message: 'Controller name?'
        }],

        // 执行的操作
        actions: [{
            type: 'add',
            path: 'src/controllers/{{dashCase name}}.js',
            templateFile: 'plop-template/controller.hbs',
        }]
    });
};

        # Inquirer.js

        Inquirer.js是Node.js命令行交互工具。vue-cli也使用了该工具。

        # 交互

        # question

        • type 问题的类型
          • input 输入内容
          • confirm 确认
          • list 列表，使用箭头上下选择
          • rawlist 列表，使用数字选择
          • expand 列表，展开所有选型，使用简写
          • checkbox 列表，多选，并且支持分隔符
          • password 输入密码
          • number 输入数字
          • editor 编辑器
        • name 存储用户回答的变量
        • message 打印出来的问题
        • default 如果用户没有输入内容，会使用默认的
        • choices 数组或者返回数组的函数
        • validate 校验输入的内容是否符合规定
        • filter 对用户的回答进行过滤处理，返回处理后的值。
        • transformer
        • when 根据前面问题的回答，判断当前问题是否需要被回答
        • pageSize 渲染行数
        • prefix 修改message默认前缀
        • suffix 修改message默认后缀
        • askAnswered 如果答案已经存在，则强制提示问题

        # 使用

        const inquirer = require("inquirer");

inquirer
  .prompt([
    {
      type: "input",
      message: "你的姓名是什么",
      name: "name",
      default: "天才",
    },
    {
      type: "number",
      message: "请输入你的年龄：",
      name: "num",
    },
    {
      type: "input",
      message: "请输入你的手机号：",
      name: "phone",
      validate: (val) => {
        if (val.match(/\d{11}/g)) {
          return true;
        }
        return "请输入11位数字";
      },
    },
    {
      type: "password",
      message: "请输入你的密码：",
      name: "pwd",
    },
    {
      type: "confirm",
      message: "是否要继续下一题？",
      name: "watch",
      default: true,
    },
    {
      type: "confirm",
      message: "是否能看到我取决于上面是否通过？",
      name: "pass",
      suffix: "后缀",
      when: function (answer) {
        return answer.watch; // 只有我return true才会这个confirm
      },
    },
    {
      type: "list",
      message: "请选择最喜欢的一个框架：",
      name: "fruit",
      default: "Vue",
      prefix: "☆☆☆☆",
      suffix: "****",
      choices: ["Vue", "React", "Angular"],
      filter: function (val) {
        // 使用filter将值变为大写
        return val.toUpperCase();
      },
    },
    {
      type: "rawlist",
      message: "请选择最喜欢的一个框架：",
      name: "fa",
      default: "Vue",
      choices: ["Vue", "React", "Angular"],
      filter: function (val) {
        // 使用filter将值变为大写
        return val.toUpperCase();
      },
    },
    {
      type: "expand",
      message: "请选择最喜欢的一个框架：",
      name: "color",
      default: "vue",
      choices: [
        {
          key: "V",
          value: "vue",
        },
        {
          key: "R",
          value: "react",
        },
        {
          key: "A",
          value: "angular",
        },
      ],
    },
    {
      type: "checkbox",
      message: "选择一个或多个框架：",
      name: "color",
      choices: [
        {
          name: "jquery",
        },
        new inquirer.Separator(), // 添加分隔符
        {
          name: "vue",
          checked: true, //默认
        },
        {
          name: "react",
        },
        {
          name: "angular",
        },
        new inquirer.Separator("--- 分隔符 ---"), // 自定义分隔符
        {
          name: "backbone",
        },
      ],
    },
    {
        type: "editor",
        message: "写下你想写的东西：",
        name: "editor",
        validate: function(text) {
            if (text.split('\n').length < 3) {
              return 'Must be at least 3 lines.';
            }
            return true;
        }
    },
  ])
  .then((answer) => {
    console.log(answer);
  });
